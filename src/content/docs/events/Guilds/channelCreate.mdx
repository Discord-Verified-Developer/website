import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Intents">

This requires the `Guilds` intent!

</Aside>

<Aside title="Functions" type="tip">

This event has a function to retrieve data:

[$newChannel[option]](/functions/newchannel)

[$oldChannel[option]](/functions/oldchannel)

</Aside>

<Example>

<div slot="default">

```js
client.channelCreateCommand({
    name: "Channel Create",
    channel: "$channelId",
    code: `
        $log[$authorId just created a new channel called: $newChannel[name]]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Channel Create",
    channel: "$channelId",
    type: "channelCreate",
    code: `
        $log[$authorId just created a new channel called: $newChannel[name]]
    `
};
```

</div>
</Example>
