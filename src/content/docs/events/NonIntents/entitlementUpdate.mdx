import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Functions">

This event has a function to retrieve data:

[$newEntitlementData[option]](/functions/newentitlementdata)

[$oldEntitlementData[option]](/functions/oldentitlementdata)

</Aside>

<Example>
<div slot="default">

```js
client.entitlementUpdateCommand({
    name: "Entitlement Update",
    channel: "$channelId",
    code: `
        $log[$oldEntitlementData[userId]'s entitlement just updated!]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Entitlement Update",
    channel: "$channelId",
    type: "entitlementUpdate",
    code: `
        $log[$oldEntitlementData[userId]'s entitlement just updated!]
    `
};
```

</div>
</Example>
