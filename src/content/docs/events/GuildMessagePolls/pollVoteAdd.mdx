---
title: onPollVoteAdd
description: This event triggers whenever a vote is added to a poll.
id: pollVoteAdd
---

import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Intents">

This requires the `GuildMessagePolls` intent!

</Aside>

<Example>

<div slot="default">

```js
client.pollVoteAddCommand({
    name: "Poll Vote Add",
    channel: "$channelId",
    code: `
        $log[$messageId just received a new vote by $authorId]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Poll Vote Add",
    channel: "$channelId",
    type: "pollVoteAdd",
    code: `
        $log[$messageId just received a new vote by $authorId]
    `
};
```

</div>
</Example>
